# Train the Model (S3)

```py
#!pip install boto3 scikit-learn

from sklearn import svm
from sklearn import datasets
from joblib import dump
import boto3
import os

minio_endpoint = 'http://minio.minio-system.svc.cluster.local:9000'  # MinIO ì—”ë“œí¬ì¸íŠ¸
minio_access_key = 'modelregistry'  # MinIO ì•¡ì„¸ìŠ¤ í‚¤
minio_secret_key = 'modelregistry1!'  # MinIO ë¹„ë°€ í‚¤
bucket_name = 'model-registry-bucket'  # ë²„í‚· ì´ë¦„
model_code = 'sklearn-iris-model'  # ëª¨ë¸ ì½”ë“œ
model_filename = 'model.joblib'  # ëª¨ë¸ íŒŒì¼ ì´ë¦„


iris = datasets.load_iris()
X, y = iris.data, iris.target

clf = svm.SVC(gamma='scale')
clf.fit(X, y)

dump(clf, model_filename)

s3 = boto3.client(
    's3',
    endpoint_url=minio_endpoint,
    aws_access_key_id=minio_access_key,
    aws_secret_access_key=minio_secret_key
)

s3.upload_file(model_filename, bucket_name, f'{model_code}/{model_filename}')

print(f"ëª¨ë¸ì´ MinIOì˜ {bucket_name}/{model_code}ì— ì €ì¥ë˜ì—ˆìŠµë‹ˆë‹¤.")

# Object Storageë¥¼ ì‚¬ìš©í•˜ê¸° ìœ„í•œ Service Account ë° Secret ìƒì„±
MINIO_API_ENDPOINT="minio.minio-system.svc.cluster.local:9000"

NAMESPACE="kubeflow-test-test-com"

cat <<EOF | tee ~/minio-sa-secret.yaml
apiVersion: v1
kind: ServiceAccount
metadata:
  name: $SERVICE_ACCOUNT_NAME
  namespace: $NAMESPACE
  annotations:
    serving.kserve.io/s3-endpoint: $MINIO_API_ENDPOINT
    serving.kserve.io/s3-usehttps: "0"
    serving.kserve.io/s3-region: "minio"
    serving.kserve.io/s3-useanoncredential: "false"

---

apiVersion: v1
kind: Secret
metadata:
  name: $SECRET_NAME
  namespace: $NAMESPACE
  annotations:
     serving.kserve.io/s3-endpoint: $MINIO_API_ENDPOINT
     serving.kserve.io/s3-usehttps: "0"
     serving.kserve.io/s3-region: "minio"
     serving.kserve.io/s3-useanoncredential: "false"
type: Opaque
stringData:
  AWS_ACCESS_KEY_ID: $MINIO_ACCESS_KEY
  AWS_SECRET_ACCESS_KEY: $MINIO_SECRET_KEY

---
apiVersion: v1
kind: ServiceAccount
metadata:
  name: $SERVICE_ACCOUNT_NAME
  namespace: $NAMESPACE
secrets:
  - name: $SECRET_NAME
EOF

kubectl apply -f ~/minio-sa-secret.yaml

kubectl get sa -n $NAMESPACE

# Model ë°°í¬

  name: $INF_MODEL_NAME
  namespace: $NAMESPACE

  predictor:
    serviceAccountName: $SERVICE_ACCOUNT_NAME
    model:
      modelFormat:
        name: $MODEL_FORMAT
      storageUri: "s3://$BUCKET_NAME/$MODEL_CODE"
EOF
```
